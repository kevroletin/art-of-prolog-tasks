* Задание 
** Условие
sum/2 с использованием plus/3 и без
** Решение 1

#+begin_src prolog
natural_number(0).
natural_number(s(X)) :- natural_number(X).

plus(A, 0, A) :-
        natural_number(A).

plus(A, s(B), s(Res)) :-
        natural_number(A),
        natural_number(B),
        sum(A, B, Res).
#+end_src

*** Исходный код

#+begin_src prolog
sum([], 0).
sum([X | Xs], Y) :-
        list(Xs),
        sum(Xs, Z),
        plus(X, Z, Y).
#+end_src
