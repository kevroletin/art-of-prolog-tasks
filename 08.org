* Задание 
** Условие
   (3.3iii) Написать программу для отношения no_doubles/2.
** Решение

*** Исходный код

#+begin_src prolog
member_([], _) :- false.
member_([X | _ ], X).
member_([_ | Xs], Y) :-
        list(Xs),
        member_(Xs, Y).

not_member([], _).
not_member([X | _ ], X) :- false.
not_member([X | Xs], Y) :-
        list(Xs),
        \=(X, Y),
        not_member(Xs, Y).
#+end_src
    
#+begin_src prolog
no_doubles([], []).
no_doubles([X | Xs], [X | Ys]) :-
        list(Xs),
        not_member(Xs, X),
        no_doubles(Xs, Ys).
no_doubles([X | Xs], Ys) :-
        list(Xs),
        member_(Xs, X),
        no_doubles(Xs, Ys).
#+end_src
